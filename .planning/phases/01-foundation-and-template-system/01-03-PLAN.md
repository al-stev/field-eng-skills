```javascript
const PAGE_TYPES = [
  { slug: 'user-journey', label: 'User Journey' },
  { slug: 'cohort-analysis', label: 'Cohort Analysis' },
  { slug: 'engagement-decay', label: 'Engagement Decay' },

  { slug: 'sdk-versions', label: 'SDK Versions' },
  { slug: 'performance', label: 'Performance' },
];

function buildNav() {
  const nav = document.getElementById('pageNav');
  const today = PAGE_DATA.generated || new Date().toISOString().slice(0, 10);
  PAGE_TYPES.forEach(p => {
    const a = document.createElement('a');
    a.className = 'nav-link' + (p.slug === PAGE_DATA.page_type ? ' active' : '');
    a.href = './' + today + '-' + p.slug + '.html';
    a.textContent = p.label;
    if (p.slug === PAGE_DATA.page_type) a.href = '#';
    nav.appendChild(a);
  });
}
```

f. **Render functions:**
```javascript
function renderHeader() {
  document.getElementById('pageTitle').textContent = PAGE_DATA.customer;
  document.getElementById('generatedDate').textContent = 'Generated ' + PAGE_DATA.generated;
  document.getElementById('dateRange').textContent = PAGE_DATA.period.start + ' \u2014 ' + PAGE_DATA.period.end;
}

function renderKPIs() {
  const row = document.getElementById('kpiRow');
  (PAGE_DATA.kpis || []).forEach(kpi => {
    const item = document.createElement('div');
    item.className = 'kpi-item';
    item.innerHTML = '<span class="kpi-value">' + kpi.value + '</span>' +
                     '<span class="kpi-label">' + kpi.label + '</span>';
    row.appendChild(item);
  });
}

function renderNarrative() {
  const container = document.getElementById('narrativeContent');
  let html = '<p>' + AI_NARRATIVE.executive_summary + '</p>';
  if (AI_NARRATIVE.highlights && AI_NARRATIVE.highlights.length) {
    html += '<h4>Key Highlights</h4><ul>' +
            AI_NARRATIVE.highlights.map(h => '<li>' + h + '</li>').join('') + '</ul>';
  }
  if (AI_NARRATIVE.recommendations && AI_NARRATIVE.recommendations.length) {
    html += '<h4>Recommendations</h4><ul>' +
            AI_NARRATIVE.recommendations.map(r => '<li>' + r + '</li>').join('') + '</ul>';
  }
  container.innerHTML = html;
}
```
